export const sizes = ['xxs', 'xs', 'sm', 'md', 'lg', 'xl', 'xxl', 'xxxl'];
export const colors = ['white', 'text-primary', 'text-secondary', 'accent-primary', 'accent-secondary'];
export const paddingSizes = ['xxs', 'xs', 'sm', 'md', 'lg', 'xl', 'xxl', 'xxxl'];

export const ColorPreview = ({ color }) =>
  color === 'white' ? (
    <GreyBox>
      <Loader size="md" color={color} />
    </GreyBox>
  ) : (
    <Loader size="md" color={color} />
  );

export const SizeMatrix = () => (
